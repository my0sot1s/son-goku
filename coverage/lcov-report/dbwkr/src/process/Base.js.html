'use strict';
 
import { config } from '../../globals';
import mongoClient from 'mongodb';
import storages from './../storages';
import { modelBase } from '../models';
 
import nconf from './../initialize';
const errCode = nconf.get('errCode');
 
const MongoClient = mongoClient.MongoClient;
/**
 *Something lke interface
 */
export class IBaseProcess {
  /**
   *collectionName:
   *_db:
   */
  constructor() {
    this.collectionName = '';
    this._db = {};// eslint-disable-line
  }
  /**
   *get Collection
   */
  get getCollection() {
    return this.collectionName;
  }
  /**
  *set Collection
  *@param :collection lay ten collection
  */
  set setCollection(collection) {
    this.collectionName = collection;
  }
}
/**
 *Implement it
 */
export default class BaseProcess extends IBaseProcess {
  /**
   *@param:default
   */
  constructor(chose) {
    super();
    this.url = config.dbGenUrl(nconf.get('db'));
    this.collectionName = modelBase[chose];
    this.chose = this.collectionName ? chose : -1;
  }
  /**
   *Open connect
   */
  open(collectionName) {
    return new Promise((resolve, reject) => { // Use connect method to connect to the Server
      try {
        MongoClient.connect(this.url, (err, db) => {
          if (err || collectionName === null || typeof collectionName !== 'string') {
            console.log(errCode.errConnection);
            reject(err || errCode.errCollectionName);
          }
          else {
            resolve(db.collection(collectionName));
          }
        });
      } catch (error) {
        throw new Error(errCode.errBase);
      }
 
    });
  }
  /**
   *Close connect
   */
  close(_db) {
    try {
      if (_db) {
        _db.close();
        return true;
      }
      else {
        return false;
      }
    } catch (error) {
      throw new Error(errCode.errBase);
    }
  }
  /**
   *reset storages in db
   * @mem={1:Product}
   */
  static resetMem(mem) {
    if (mem === undefined || typeof mem !== 'object') {
      Object.keys(storages).map(item => {
        storages[item].cache = {};
        storages[item].maxKey = 0;
      });
    }
    else {
      Object.keys(mem).map(key => {
        storages[mem[key]].cache = {};
        storages[mem[key]].maxKey = 0;
      });
    }
  }
}
