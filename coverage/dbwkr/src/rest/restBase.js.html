  Read,
  Create,
  Update,
  Deleted,
  Find,
  BaseProcess,
} from '../process';
import nconf from './../initialize';
const errCode = nconf.get('errCode');
import { config } from '../../globals';// const config = require('../configurations/config');
import express from 'express';// const express = require('express');
const router = express.Router();
 
router.all('*', config.httpMiddleWare.checkall, config.httpMiddleWare.middleware)
  /**

  .post('/reset', (req, res) => {
    try {
      console.log('--Reset:', req.body);
      BaseProcess.resetMem(req.body);
      res.status(200).send('ok');
    } catch (error) {
      res.status(errCode.errCode403).end();
    }
 
  })
  .get('/:id', (req, res) => {
    var reader = new Read(Number(req.params.id));
    reader.read().then(docs => {
      if (docs !== null) {
        console.log('--Count Items:', Object.keys(docs).length);
        res.status(200).json(docs || 200);
      }
      else {
        res.status(errCode.errCode403).end();
      }
    }).catch(error => {
      console.log(`+++ Error in http machine get ${error}`);
      res.status(errCode.errCode503).end(error);
    });
  })
  .get('/:id/id/:_id', (req, res) => {
    var finder = new Find(Number(req.params.id));
    finder.find({ _id: Number(req.params._id) })
      .then(docs => {//eslint-disable-line
        if (docs !== null) {
          res.status(200).json(docs || 200);
        }
        else {
          res.status(errCode.errCode403).end();
        }
      })
      .catch(error => {
        console.log(`+++ Error in http machine get ${error}`);
        res.status(errCode.errCode503).end(error);
      });
  })
  .post('/:id', (req, res) => {

        if (docs !== null) {
          res.status(200).json(docs || 200);
        }

          res.status(errCode.errCode403).end();
