'use strict';
 
import { config } from '../../configs';// const config = require('../configurations/config');
import http from 'http';// const http = require("http");
 
const checkWeather = (city, cb, type) => {
  var key = config.weather.key;
  http.get(`http://api.openweathermap.org/data/2.5/${type}?q=${city}&mode=json&units=metric&${key}`, response => {
    var buffer = '';
    var data = '';
 
    if (response.statusCode === 502) {
      cb('+++ loi', `Không tìm được thành phố/quốc gia`);// eslint-disable-line
    } else {
      response.on('error', e => console.log(`+++ api Provider error or can not connect:${e}`));
      response.on('data', chunk => {
        buffer += chunk;
      });
      response.on('end', err => {
        data = JSON.parse(buffer);
        if (err) {
          return cb(null, null);
        }
        if (type == config.weather.weatherType.weather) {
          var result = {}; result.data = data; result.dataText = `---Thời tiết:${city}\nNhiệt độ:${data.main.temp} *C--Gió: ${data.wind.speed} m/s 
            \nMây che trời:${data.clouds.all}%--Mô tả:${data.weather[0].description}`;
          console.warn(result.dataText);
        }
        cb(null, data);
      });
    }
  });
};
export default checkWeather;
