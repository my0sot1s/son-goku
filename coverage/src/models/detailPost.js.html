/**
 *Model class create
 */
export default class detailPost {
  /**
   *@param:json data
   */
  constructor(json) {
    this._id = Number(json._id);
    this.banner = json.banner || [];
    this.content = json.content.split('\n');
    this.images = json.images || [];
  }
  /**
   *format json
   */
  get jsondata() {
    return {
      _id: this._id,
      banner: this.banner,
      content: this.content,
      images: this.images,
    };
  }
  /**
   *return id
   */
  get id() {
    return this._id;
  }
}
